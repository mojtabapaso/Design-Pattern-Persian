```python
class Component:
    def operation(self):
        pass

class Leaf(Component):
    def operation(self):
        print("Leaf")

class Composite(Component):
    def __init__(self):
        self.children = []

    def add(self, child):
        self.children.append(child) 

    def remove(self, child):
        self.children.remove(child)

    def operation(self):
        for child in self.children:
            child.operation()

leaf1 = Leaf()
leaf2 = Leaf()

composite = Composite()
composite.add(leaf1)
composite.add(leaf2)

composite.operation()
```
